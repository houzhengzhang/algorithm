* ##### 欧拉筛
  任何合数都能表示成一系列素数的积。然后利用了每个合数必有一个最小素因子，每个合数仅被它的最小素因子筛去正好一次。所以为线性时间O(n)。

  ```c++
  #include<cstdio>
  using namespace std;
  const int P=1000000;
  int a[P];
  int main()
  {
      int n,num=0,i,j;
      scanf("%d",&n);
  
      for(i=2;i<n;++i)
      {
          // a[i] 未被筛到 则i是素数
          if(!a[i]) a[num++]=i;
          for(j=2;j*i<n;++j)
          {
              a[i*j]=1;
              // i%j==0 找到j是最小的质因子
              // 保证一个数只会被最小素因子筛到一次
              //eg 3*5=5*3=15
              // 如果没有判断 15 会被筛到两次
              if(i%j==0) break;
          }
      }
      for(i=0;i<num;++i)
          printf("%d\n",a[i]);
  
      return 0;
  }
  ```
